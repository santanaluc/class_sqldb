SQL:
	DDL = createm alter, drop, truncate - mais utilizada
	DML = update, insert, delete, merge - mais utilizada
	TCL = commit, rollback, savepoint
	DCL = Grant, Revoke

Tipos de caracteres:
	Varchar(n) - quando nao se sabe o tamanho fixo de um campo
		Só armazena o necessário -- consome muito recurso, deixando o sistema mais lento
	char(n) - define uma quantidade fixa
		Utiliza quando sabe que sempre vai ser um valor

	Bigint 
	int
	smallint
	tinyint
	bit 

	decimal(casas inteiras, casas decimais)
	numérico(casas inteiras, casas decimais)

	money
	datetime = data e hora

Comando "Create":
	CREATE DATABASE <nome-do-banco> 

	CREATE TABLE <nome-tabela> (<coluna> <tipo>)

	CREATE [UNIQUE] INDEX <nome-indice> on <nome-tabela> (<coluna> [asc/desc])
		criar um indice

Comando DROP:
	DROP DATABASE <nome-do-banco>

	DROP TABLE <nome-da-tabela>

Comando ALTER:
	ALTER TABLE <nome-da-tabela>
		ADD PRIMARY KEY (coluna) -- chave primária

	ALTER TABLE <nome-da-tabela>
		ADD FOREIGN KEY (<coluna>) -- chave estrangeira
		REFERENCES <tabela> 

	ALTER TABLE <nome-da-tabela>
		DROP COLUMN (<coluna>)

QUANDO FOR CHAVE PRIMÁRIA, COLOCAR: "not null"
